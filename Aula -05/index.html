      def remove_outlier(df_old):
             #Tratando Outlier 
             #https://medium.com/turing-talks/como-fazer-uma-limpeza-de-dados-completa-em-python-7abc9dfc19b8
            
             df_old.reset_index(drop=True,inplace=True )
            
             #remove duplicate 
             df_old.drop_duplicates(keep='first',inplace=True)
            
            
              #  df_old.quantile(0.25),  df_old.quantile(0.75), df_old['TEMPERATURA_DOMO_INICIO'].quantile(0.25)
            
            
             #Step-5: Finding the IQR
             percentile25 = df_old['TEMPERATURA_DOMO_INICIO'].quantile(0.25)
             percentile75 = df_old['TEMPERATURA_DOMO_INICIO'].quantile(0.75)
            
             iqr = percentile75 - percentile25
            
            
             #Step-6: Finding the upper and lower limits
             upper_limit = percentile75 + 1.5 * iqr
             lower_limit = percentile25 - 1.5 * iqr
            
             #Step-7: Finding outliers
             df_outlier = df_old[(df_old['TEMPERATURA_DOMO_INICIO'] > upper_limit) | (df_old['TEMPERATURA_DOMO_INICIO'] < lower_limit)]
             
            
             #Step-8: Trimming outliers
             new_df_date = df_old.drop(df_outlier.index)
            
             
             new_df_date.index = range(1, len(new_df_date) + 1)
             new_df_date.reset_index(drop=True)
            
             #new_df_date.to_csv('../IFES/Export/hs_dados-new.csv')
            
            
            return new_df_date

            # Função para calcular o fatorial
            def fatorial(n):
                if n == 0 or n == 1:
                    return 1
                else:
                    return n * fatorial(n - 1)

            # Solicita ao usuário um número
            numero = int(input("Digite um número para calcular o fatorial: "))

            # Calcula e exibe o fatorial
            resultado = fatorial(numero)
            print(f"O fatorial de {numero} é {resultado}.")
